# NSWFparse

### Installation

```shell
npm instal nswfparse
```

## Example

```js
const nswfparse = require("nswfparse");

// Rule34
const options = {
  tags: ["toys", "forest"],
  numPage: 1,
  limit: 1, // limit per request, maximum 100
  remove_empty: true,
  parse_tags: true, // whether to use tags
  random: false, // get random post
};
nswfparse
  .r34(options)
  .then((payload) => {
    console.log(payload);
  })
  .catch((error) => {
    console.error(error);
  });

// Reddit
nswfparse.reddit.real
  .pov()
  .then((payload) => {
    console.log(payload);
  })
  .catch((error) => {
    console.error(error);
  });
// Or
nswfparse
  .redditCustom(["cats"])
  .then((payload) => {
    console.log(payload);
  })
  .catch((error) => {
    console.error(error);
  });
```
